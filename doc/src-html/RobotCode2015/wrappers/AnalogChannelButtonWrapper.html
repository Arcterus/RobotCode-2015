001package RobotCode2015.wrappers;
002
003import edu.wpi.first.wpilibj.AnalogChannel;
004
005/**
006 *
007 * @author 1072
008 */
009public class AnalogChannelButtonWrapper extends ButtonWrapper {
010
011    private static double THRESHOLD = 0.5;
012    private final int port;
013    private AnalogChannel channel;
014
015    public AnalogChannelButtonWrapper(int port) {
016        this.port = port;
017        channel = new AnalogChannel(port);
018    }
019
020    public boolean get() {
021        channel.updateTable();
022        return channel.getValue() > THRESHOLD;
023    }
024    
025    public void setThreshold(double newVal) { 
026        THRESHOLD = newVal; 
027    }
028    
029    public double getThreshold() {
030        return THRESHOLD;
031    }
032}

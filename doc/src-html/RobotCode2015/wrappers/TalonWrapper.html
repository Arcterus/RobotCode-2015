001package RobotCode2015.wrappers;
002
003import edu.wpi.first.wpilibj.Talon;
004
005/**
006 *
007 * @author atierno
008 */
009public class TalonWrapper extends Talon{
010    
011    private boolean isReversed = false;
012    
013    /**
014     * Initializes a TalonWrapper with its channel.
015     * @param channel The PWM channel on the digital module the Talon is attached to.
016     */
017    public TalonWrapper (int channel) {
018        super(channel);
019    }
020    
021    /**
022     * Initializes a TalonWrapper with its channel and reversed flag.
023     * @param channel The PWM channel on the digital module the Talon is attached to.
024     * @param isReversed Whether or not the output of this Talon should be flipped.
025     */
026    public TalonWrapper (int channel, boolean isReversed) {
027        super(channel);
028        this.isReversed = isReversed;
029    }
030    
031    /**
032     * Initializes a TalonWrapper with its channel and slot.
033     * @param channel The PWM channel on the digital module the Talon is attached to.
034     * @param slot The slot on the chassis the digital module is plugged into.
035     */
036    public TalonWrapper (int channel, int slot) {
037        super(slot, channel);
038    }
039    
040    /**
041     * Initializes a TalonWrapper with its channel, slot, and reversed flag
042     * @param channel The PWM channel on the digital module the Talon is attached to.
043     * @param slot The slot on the chassis the digital module is plugged into.
044     * @param isReversed Whether or not the output of this Talon should be flipped.
045     */
046    public TalonWrapper (int channel, int slot, boolean isReversed) {
047        super(slot, channel);
048        this.isReversed = isReversed;
049    }
050    
051    /**
052     * Sets the speed of the Talon (handles reversing).
053     * @param speed The speed to set the Talon.
054     */
055    public void set (double speed) {
056        super.set(isReversed ? -speed : speed);
057    }
058    
059    /**
060     * Sets the reversed flag of the Talon.
061     * @param flag Set whether or not to reverse the Talon output.
062     */
063    public void setReversed (boolean flag) {
064        isReversed = flag;
065    }
066    
067    /**
068     * Gets the reversed flag of the Talon.
069     * @return Whether or not the Talon output is reversed
070     */
071    public boolean getReversed () {
072        return isReversed;
073    }
074}

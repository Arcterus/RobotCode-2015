001package RobotCode2015.wrappers;
002import edu.wpi.first.wpilibj.GenericHID;
003
004/**
005 *
006 * @author Manan
007 */
008public class JoystickButtonWrapper extends ButtonWrapper{
009    private final GenericHID m_joystick;
010    private final int m_buttonNumber;
011
012    /**
013     * Create a joystick button for triggering commands
014     * @param joystick The GenericHID object that has the button (e.g. Joystick, KinectStick, etc)
015     * @param buttonNumber The button number (see {@link GenericHID#getRawButton(int) }
016     */
017    public JoystickButtonWrapper(GenericHID joystick, int buttonNumber) {
018        m_joystick = joystick;
019        m_buttonNumber = buttonNumber;
020    }
021    
022    /**
023     * Gets the value of the joystick button
024     * @return The value of the joystick button
025     */
026    public boolean get() {
027        return m_joystick.getRawButton(m_buttonNumber);
028    }
029}

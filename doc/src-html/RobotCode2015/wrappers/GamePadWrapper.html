002package RobotCode2015.wrappers;
003
004import edu.wpi.first.wpilibj.Joystick;
005
006/**
007 * A Joystick wrapper for gamepads that include more accurate/useful names for the analogue stick axes
008 * @author neymikajain
009 */
010public class GamePadWrapper extends Joystick {
011
012    public GamePadWrapper(int port) {
013        super(port);
014    }
015
016    public double getLeftX() {
017        return super.getX();
018    }
019
020    public double getLeftY() {
021        return -super.getY(); //by default, forward returns a negative number, which is unintuitive
022    }
023
024    public double getRightX() {
025        return super.getZ();
026    }
027
028    public double getRightY() {
029        return -super.getThrottle(); //by default, forward returns a negative number, which is unintuitive
030    }
031}

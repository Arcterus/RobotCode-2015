002package RobotCode2015.wrappers;
003
004import RobotCode2015.RobotMap;
005import edu.wpi.first.wpilibj.Joystick;
006
007/**
008 * A Joystick wrapper for gamepads that include more accurate/useful names for the analogue stick axes
009 * @author neymikajain
010 */
011public class GamePadWrapper extends Joystick {
012    private final JoystickButtonWrapper buttonA;
013    private final JoystickButtonWrapper buttonB;
014    private final JoystickButtonWrapper buttonX;
015    private final JoystickButtonWrapper buttonY;
016    private final JoystickButtonWrapper buttonStart;
017    private final JoystickButtonWrapper buttonSelect;
018    private final JoystickButtonWrapper buttonStickLeft;
019    private final JoystickButtonWrapper buttonStickRight;
020    private final JoystickButtonWrapper buttonBumperLeft;
021    private final JoystickButtonWrapper buttonBumperRight;
022    private final JoystickButtonWrapper buttonTriggerLeft;
023    private final JoystickButtonWrapper buttonTriggerRight;
024
025    
026    public GamePadWrapper(int port) {
027        super(port);
028        buttonA = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_A);
029        buttonB = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_B);
030        buttonX = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_X);
031        buttonY = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_Y);
032        buttonStart = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_START);
033        buttonSelect = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_SELECT);
034        buttonStickLeft = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_STICK_LEFT);
035        buttonStickRight = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_STICK_RIGHT);
036        buttonBumperLeft = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_BUMPER_LEFT);
037        buttonBumperRight = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_BUMPER_RIGHT);
038        buttonTriggerLeft = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_TRIGGER_LEFT);
039        buttonTriggerRight = new JoystickButtonWrapper(this, RobotMap.Gamepad.BUTTON_TRIGGER_RIGHT);
040    }
041
042    public double getLeftX() {
043        return super.getX();
044    }
045
046    public double getLeftY() {
047        return -super.getY(); //by default, forward returns a negative number, which is unintuitive
048    }
049
050    public double getRightX() {
051        return super.getZ();
052    }
053
054    public double getRightY() {
055        return -super.getThrottle(); //by default, forward returns a negative number, which is unintuitive
056    }
057    
058    /**
059     * Gets whether or not Button A is pressed
060     * @return The state of the button
061     */
062    public boolean getButtonAState() {
063        return buttonA.get();
064    }
065    
066    /**
067     * Gets whether or not Button B is pressed
068     * @return The state of the button
069     */
070    public boolean getButtonBState() {
071        return buttonB.get();
072    }
073    
074    /**
075     * Gets whether or not Button X is pressed
076     * @return The state of the button
077     */
078    public boolean getButtonXState() {
079        return buttonX.get();
080    }
081    
082    /**
083     * Gets whether or not Button Y is pressed
084     * @return The state of the button
085     */
086    public boolean getButtonYState() {
087        return buttonY.get();
088    }
089    
090    /**
091     * Gets whether or not the Start Button is pressed
092     * @return The state of the button
093     */
094    public boolean getButtonStartState() {
095        return buttonStart.get();
096    }
097    
098    /**
099     * Gets whether or not the Select Button is pressed
100     * @return The state of the button
101     */
102    public boolean getButtonSelectState() {
103        return buttonSelect.get();
104    }
105    
106    /**
107     * Gets whether or not the Left Stick Button is pressed
108     * @return The state of the button
109     */
110    public boolean getButtonStickLeftState() {
111        return buttonStickLeft.get();
112    }
113    
114    /**
115     * Gets whether or not the Right Stick Button is pressed
116     * @return The state of the button
117     */
118    public boolean getButtonStickRightState() {
119        return buttonStickRight.get();
120    }
121    
122    /**
123     * Gets whether or not the Left Bumper is pressed
124     * @return The state of the button
125     */
126    public boolean getButtonBumperLeftState() {
127        return buttonBumperLeft.get();
128    }
129    
130    /**
131     * Gets whether or not the Right Bumper is pressed
132     * @return The state of the button
133     */
134    public boolean getButtonBumperRightState() {
135        return buttonBumperRight.get();
136    }
137    
138    /**
139     * Gets whether or not the Left Trigger is pressed
140     * @return The state of the button
141     */
142    public boolean getButtonTriggerLeftState() {
143        return buttonTriggerLeft.get();
144    }
145    
146    /**
147     * Gets whether or not the Right Trigger is pressed
148     * @return The state of the button
149     */
150    public boolean getButtonTriggerRightState() {
151        return buttonTriggerRight.get();
152    }
153    
154    /**
155     * Gets whether or not the Right Button on the DPad is pressed. Since the DPad
156     * is an axis, the code checks to see if the button is pushed significantly far enough
157     * to the right.
158     * @return The state of the button
159     */
160    public boolean getDPadRightState() {
161        return this.getRawAxis(RobotMap.Gamepad.AXIS_DPAD) > .5;
162    }
163    
164    /**
165     * Gets whether or not the Right Button on the DPad is pressed. Since the DPad
166     * is an axis, the code checks to see if the button is pushed significantly far enough
167     * to the left.
168     * @return The state of the button
169     */
170    public boolean getDPadLeftState() {
171        return this.getRawAxis(RobotMap.Gamepad.AXIS_DPAD) < -.5;
172    }
173    
174    /**
175     * Gets an instance of Button A
176     * @return An instance of the button
177     */
178    public JoystickButtonWrapper getButtonA() {
179        return buttonA;
180    }
181    
182    /**
183     * Gets an instance of Button B
184     * @return An instance of the button
185     */
186    public JoystickButtonWrapper getButtonB() {
187        return buttonB;
188    }
189    
190    /**
191     * Gets an instance of Button X
192     * @return An instance of the button
193     */
194    public JoystickButtonWrapper getButtonX() {
195        return buttonX;
196    }
197    
198    /**
199     * Gets an instance of Button Y
200     * @return An instance of the button
201     */
202    public JoystickButtonWrapper getButtonY() {
203        return buttonY;
204    }
205    
206    /**
207     * Gets an instance of the Start Button
208     * @return An instance of the button
209     */
210    public JoystickButtonWrapper getButtonStart() {
211        return buttonStart;
212    }
213    
214    /**
215     * Gets an instance of the Select Button
216     * @return An instance of the button
217     */
218    public JoystickButtonWrapper getButtonSelect() {
219        return buttonSelect;
220    }
221    
222    /**
223     * Gets an instance of the Left Stick Button
224     * @return An instance of the button
225     */
226    public JoystickButtonWrapper getButtonStickLeft() {
227        return buttonStickLeft;
228    }
229    
230    /**
231     * Gets an instance of the Right Stick Button
232     * @return An instance of the button

234    public JoystickButtonWrapper getButtonStickRight() {
235        return buttonStickRight;

243        return buttonBumperLeft;
244    }

247     * Gets an instance of the Right Bumper
248     * @return An instance of the button
249     */
250    public JoystickButtonWrapper getButtonBumperRight() {
251        return buttonBumperRight;
252    }
253    
254    /**
255     * Gets an instance of the Left Trigger
256     * @return An instance of the button
257     */
258    public JoystickButtonWrapper getButtonTriggerLeft() {
259        return buttonTriggerLeft;
260    }
261    
262    /**
263     * Gets an instance of the Right Trigger
264     * @return An instance of the button
265     */
266    public JoystickButtonWrapper getButtonTriggerRight() {
267        return buttonTriggerRight;
268    }

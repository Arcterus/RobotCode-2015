003/**
004 * An immutable implementation of a Matrix in Java
005 * Primary logic and Matrix code derived and/or reproduced from
006 *  <a href="http://introcs.cs.princeton.edu/java/95linear/Matrix.java.html">
007 *      Princeton Introduction to Programming in Java</a>
008 *
009 * @author Robert Sedgewick
010 * @author Kevin Wayne
011 * @author Manan
012 */
013public class Matrix {
014    /**
015     * The number of rows.
016     */
017    private final int rows;
018
019    /**
020     * The number of columns
021     */
022    private final int cols;
023
024    /**
025     * A 2-dimensional array containing the elements of the matrix.
026     */
027    private final double[][] data;
028
029    /**
030     * Create an M-by-N matrix of zeros
031     * @param M The number of rows
032     * @param N The number of columns
033     */
034    public Matrix(int M, int N) {
035        this.rows = M;
036        this.cols = N;
037        data = new double[M][N];
038    }
039
040    /**
041     * Create a matrix based on a 2-dimensional array
042     * @param data The array to create a matrix of
043     */
044    public Matrix(double[][] data) {
045        rows = data.length;
046        cols = data[0].length;
047        this.data = new double[rows][cols];
048        for (int i = 0; i < rows; i++)
049            System.arraycopy(data[i], 0, this.data[i], 0, cols);
050    }
051
052    /**
053     * Create a new matrix that is a copy of another one
054     * @param A The matrix to copy
055     */
056    public Matrix(Matrix A) { this(A.data); }
057
058
059    /**
060     * Gets the value of an element in the matrix
061     * @param x The row number
062     * @param y The column number
063     * @return The value at index (x,y)
064     */
065    public double getValue(int x, int y) {
066        return data[x][y];
067    }
068
069    /**
070     * Sets the value of an element in the matrix
071     * @param x The row number
072     * @param y The column number
073     * @param val The value to set index (x,y) to
074     */
075    public void setValue(int x, int y, double val) {
076        data[x][y] = val;
077    }
078
079    /**
080     * Create a new N-by-N identity matrix
081     * @param N The size of the matrix
082     * @return An identity matrix of size N
083     */
084    public static Matrix identity(int N) {
085        Matrix I = new Matrix(N, N);
086        for (int i = 0; i < N; i++)
087            I.data[i][i] = 1;
088        return I;
089    }
090
091    /**
092     * Swap rows i and j
093     * @param i The first row to swap
094     * @param j The row to swap with
095     */
096    private void swap(int i, int j) {
097        double[] temp = data[i];
098        data[i] = data[j];
099        data[j] = temp;
100    }
101
102    /**
103     * Creates the transpose matrix, A<sup>T</sup>
104     * @return The transpose of a given matrix
105     */
106    public Matrix transpose() {
107        Matrix A = new Matrix(cols, rows);
108        for (int i = 0; i < rows; i++)
109            for (int j = 0; j < cols; j++)
110                A.data[j][i] = data[i][j];
111        return A;
112    }
113
114    /**
115     * Adds the matrix to another one
116     * @param B The matrix to add to
117     * @return The sum of the two matrices
118     */
119    public Matrix plus(Matrix B) {
120        Matrix A = this;
121        if (B.rows != A.rows || B.cols != A.cols) throw new RuntimeException("Illegal matrix dimensions.");
122        Matrix C = new Matrix(rows, cols);
123        for (int i = 0; i < rows; i++)
124            for (int j = 0; j < cols; j++)
125                C.data[i][j] = A.data[i][j] + B.data[i][j];
126        return C;
127    }
128
129
130    /**
131     * Takes the difference of the matrix and another one
132     * @param B The matrix to subtract this one from
133     * @return The difference of the two matrices
134     */
135    public Matrix minus(Matrix B) {
136        Matrix A = this;
137        if (B.rows != A.rows || B.cols != A.cols) throw new RuntimeException("Illegal matrix dimensions.");
138        Matrix C = new Matrix(rows, cols);
139        for (int i = 0; i < rows; i++)
140            for (int j = 0; j < cols; j++)
141                C.data[i][j] = A.data[i][j] - B.data[i][j];
142        return C;
143    }
144
145    /**
146     * Takes the product of this matrix and another
147     * @param B The matrix to multiply by
148     * @return The product of the matrix and the other
149     */
150    public Matrix mult(Matrix B) {
151        Matrix A = this;
152        if (A.cols != B.rows) throw new RuntimeException("Illegal matrix dimensions.");
153        Matrix C = new Matrix(A.rows, B.cols);
154        for (int i = 0; i < C.rows; i++)
155            for (int j = 0; j < C.cols; j++)
156                for (int k = 0; k < A.cols; k++)
157                    C.data[i][j] += (A.data[i][k] * B.data[k][j]);
158        return C;
159    }
160
161    /**
162     * Solves the equation A * x = b for x
163     * @param rhs The right hand side of the equation, b.
164     * @return x = A<sup>-1</sup> * b, assuming A is square and has full rank
165     */
166    public Matrix solve(Matrix rhs) {
167        if (rows != cols || rhs.rows != cols)// || rhs.cols != 1)
168            throw new RuntimeException("Illegal matrix dimensions.");
169
170        // create copies of the data
171        Matrix A = new Matrix(this);
172        Matrix b = new Matrix(rhs);
173
174        // Gaussian elimination with partial pivoting
175        for (int i = 0; i < cols; i++) {
176            int max = i;
177            for (int j = i + 1; j < cols; j++)
178                if (Math.abs(A.data[j][i]) > Math.abs(A.data[max][i]))
179                    max = j;
180            A.swap(i, max);
181            b.swap(i, max);
182
183            // singular
184            if (A.data[i][i] == 0.0) throw new RuntimeException("Matrix is singular.");
185
186            // pivot within b
187            for (int j = i + 1; j < cols; j++)
188                b.data[j][0] -= b.data[i][0] * A.data[j][i] / A.data[i][i];
189
190            // pivot within A
191            for (int j = i + 1; j < cols; j++) {
192                double m = A.data[j][i] / A.data[i][i];
193                for (int k = i+1; k < cols; k++) {
194                    A.data[j][k] -= A.data[i][k] * m;
195                }
196                A.data[j][i] = 0.0;
197            }
198        }
199
200        // back substitution
201        Matrix x = new Matrix(cols, 1);
202        for (int j = cols - 1; j >= 0; j--) {
203            double t = 0.0;
204            for (int k = j + 1; k < cols; k++)
205                t += A.data[j][k] * x.data[k][0];
206            x.data[j][0] = (b.data[j][0] - t) / A.data[j][j];
207        }
208        return x;
209    }

211    /**
212     * Creates a new {@link Vector3D} with the first column of the matrix.
213     * @return A new {@link Vector3D} based on the first column of the matrix.
214     */
215    public Vector3D ToVector3DByCol() {
216        return new Vector3D(getValue(0, 0), getValue(1, 0), getValue(2, 0));
217    }
218
219    /**
220     * Creates a new {@link Vector3D} with a specified column of the matrix.
221     * @param col The column to create a vector of.
222     * @return A new {@link Vector3D} based on the given column of the matrix.
223     */
224    public Vector3D ToVector3DByCol(int col) {
225        return new Vector3D(getValue(0, col), getValue(1, col), getValue(2, col));
226    }
227
228    /**
229     * Creates a new {@link Vector3D} with the first column of the matrix, scaled by the fourth element in that column.
230     * @return A new {@link Vector3D} based on the given column of the matrix and its scale factor.
231     */
232    public Vector3D ToVector3DByColWithScale() {
233        return new Vector3D(getValue(0, 0) / getValue(3, 0), getValue(1, 0) / getValue(3, 0), getValue(2, 0) / getValue(3, 0));
234    }
235
236    /**
237     * Creates a new {@link Vector3D} with the first row of the matrix.
238     * @return A new {@link Vector3D} based on the first row of the matrix.
239     */
240    public Vector3D ToVector3DByRow() {
241        return new Vector3D(getValue(0, 0), getValue(0, 1), getValue(0, 2));
242    }
243
244    /**
245     * Creates a new {@link Vector3D} with a specified row of the matrix.
246     * @param row The row to create a vector of.
247     * @return A new {@link Vector3D} based on the given row of the matrix.
248     */
249    public Vector3D ToVector3DByRow(int row) {
250
251        return new Vector3D(getValue(row, 0), getValue(row, 1), getValue(row, 2));
252    }
253
254    /**
255     * Creates a new matrix from a Vector3D
256     * @deprecated Instead, use {@link Vector3D#toMatrix()}
257     * @param v The vector to create a matrix from
258     * @return The new matrix
259     */
260    public static Matrix FromVector3D(Vector3D v) {
261        return new Matrix(new double[][] { {v.getX()}, {v.getY()}, {v.getZ()}, {1}});
262    }
263
264    /**
265     * Checks if two matrices are equivalent
266     * @param B The matrix to compare to
267     * @return <code>true</code> if the dimensions match and the individual elements are the same.
268     */
269    public boolean equals(Matrix B) {
270        Matrix A = this;
271        if (B.rows != A.rows || B.cols != A.cols) throw new RuntimeException("Illegal matrix dimensions.");
272        for (int i = 0; i < rows; i++)
273            for (int j = 0; j < cols; j++)
274                if (A.data[i][j] != B.data[i][j]) return false;
275        return true;
276    }
277
278    /**
279     * Returns a printable form of the matrix.
280     * @return The formatted string version of the matrix
281     */
282    public String toString() {
283        String s = "";
284        for (int i = 0; i < rows; i++) {
285            for (int j = 0; j < cols; j++)
286                s += data[i][j] + " ";
287            s += "\n";
288        }
289        return s;
290    }

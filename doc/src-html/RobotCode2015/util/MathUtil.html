035    /**
036     * Returns x^power, works for any values of x and power
037     * @param x the base
038     * @param power the exponent
039     * @return x^power
040     * @author Ben H. 
041     */
042    public static double power(double x, int power) {
043        // This is a more efficient method of doing power than 1 at a time for large exponents
044        // Runs in O(log2(power)) time  
045        
046        // If power is a multiple of 2 squares x and halves it. If not it decrements it and
047        // stores what that 1 should have represented in this number. It is multiplied
048        // back in at the end.
049        double nonPowerOfTwoMultiplier = 1;
050        while (power > 1) {
051            if (power % 2 == 1) {
052                nonPowerOfTwoMultiplier *= x;
053                power--;
054            }
055            x*=x;
056            power/=2;
057        }
058        return x * nonPowerOfTwoMultiplier;
059    }

061    public static double solveLinear(double xCoeff, double  constant) {
062        // mx+b = 0 -> x = -b/m
063        if (xCoeff == 0) {
064            IOUtil.warn("Tried to solve " + constant + " == 0 for x");
065            return 0;
066        }
067        return -constant / xCoeff;
068    }

070    public static double solveQuadraticHigh(double squareCoeff, double xCoeff, double constant) {
071        // Uses quadratic formula; returns the higher one
072        // x = ( -b + sqrt(b^2-4ac) ) / 2a
073        if (squareCoeff == 0) {
074            IOUtil.println("Tried to solve a quadratic with no square term. Using linear solver");
075            return solveLinear(xCoeff, constant);
076        }
077        double determinant = power(xCoeff, 2) - (4 * squareCoeff * constant);
078        if (determinant < 0) {
079            IOUtil.warn("Tried to solve a quadratic with no real solution");
080            return 0; // is it possible to return NaN?
081        }
082        double sqrtTerm = Math.sqrt(determinant);
083        return ((sqrtTerm - xCoeff) / (2 * squareCoeff));
084    }
085    
086    public static double solveQuadraticLow(double squareCoeff, double xCoeff, double constant) {
087        // Uses quadratic formula; returns the lower one
088        // x = ( -b - sqrt(b^2-4ac) ) / 2a
089        if (squareCoeff == 0) {
090            IOUtil.println("Tried to solve a quadratic with no square term. Using linear solver");
091            return solveLinear(xCoeff, constant);
092        }
093        double determinant = power(xCoeff, 2) - (4 * squareCoeff * constant);
094        if (determinant < 0) {
095            IOUtil.warn("Tried to solve a quadratic with no real solution");
096            return 0; // is it possible to return NaN?
097        }
098        double sqrtTerm = Math.sqrt(determinant);
099        return ((-sqrtTerm - xCoeff) / (2 * squareCoeff));
100    }

102    /**
103     * Returns the standard deviation of the data set
104     * @param points
105     * @return stdev of data
106     */
107    public static double stdev(double[] points)
108    {
109         double mean = mean(points);
110         int sum = 0;
111         for(int i=0; i<points.length; i++)
112         {
113             points[i] = power(points[i]-mean, 2);
114             sum += points[i];
115         }
116         sum /= points.length;
117         return Math.sqrt(sum);
118    }

120    /**
121     * Returns the mean of the data set 
122     * @param dataSet
123     * @return the mean
124     */
125    public static double mean (double[] dataSet) {
126        double sum = 0;
127        for (int i = 0; i < dataSet.length; i++) {
128            sum += dataSet[i];
129        }
130        return (sum/dataSet.length);
131    }

133    /**
134     * Analyzes data and outputs an array of results
135     *  Uses logic from the Princeton University Libraries:
136     *  http://introcs.cs.princeton.edu/java/97data/LinearRegression.java.html
137     * @param x the x coordinates of the data
138     * @param y the y coordinates of the data
139     * @return an array that contains
140     *  Index 0: R^2 value
141     *  Index 1: beta 1 error value
142     *  Index 2: beta 0 error value
143     *  Index 3: yybar value
144     *  Index 4: residual sum of squares
145     *  Index 5: regression sum of squares
146     */
147    public static double[] linReg(double[] x, double[] y)
148    {
149        int n = 0;
150            // first pass: read in data, compute xbar and ybar
151        double sumx = 0.0, sumy = 0.0, sumx2 = 0.0;
152        double xbar = sumx / n;
153        double ybar = sumy / n;
154
155        // second pass: compute summary statistics
156            double xxbar = 0.0, yybar = 0.0, xybar = 0.0;
157            for (int i = 0; i < n; i++) 
158            {
159                    xxbar += (x[i] - xbar) * (x[i] - xbar);
160                    yybar += (y[i] - ybar) * (y[i] - ybar);
161                    xybar += (x[i] - xbar) * (y[i] - ybar);
162            }
163            double beta1 = xybar / xxbar;
164            double beta0 = ybar - beta1 * xbar;
165
166            // print results
167            //System.out.println("y   = " + beta1 + " * x + " + beta0);
168
169            // analyze results
170            int df = n - 2;
171            double rss = 0.0;      // residual sum of squares
172            double ssr = 0.0;      // regression sum of squares
173            for (int i = 0; i < n; i++) 
174            {
175                    double fit = beta1*x[i] + beta0;
176                    rss += (fit - y[i]) * (fit - y[i]);
177                    ssr += (fit - ybar) * (fit - ybar);
178            }
179
180            double R2    = ssr / yybar;
181            double svar  = rss / df;
182            double svar1 = svar / xxbar;
183            double svar0 = svar/n + xbar*xbar*svar1;
184
185            double[] arr = {R2, Math.sqrt(svar1), Math.sqrt(((svar *sumx2) / (n * xxbar))), yybar, rss, ssr}; 
186            return arr;
187     }

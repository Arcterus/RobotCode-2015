001package RobotCode2015.util;
002
003import RobotCode2015.util.MathUtil;
004
005/**
006 * Represents a three-dimensional vector 
007 * @author Manan Shah
008 */
009public class Vector3D
010{
011    /**
012     * The X-position of the point.
013     * This represents movement along the axis from one alliance side of the
014     * field towards the other alliance side of the field.
015     */
016    private double X;
017
018    /**
019     * The Y-position of the point.
020     * This represents movement along the axis from the sides of the field,
021     * parallel to the alliance walls and game objectives.
022     */
023    private double Y;
024
025    /**
026     * The Z-position of the point.
027     * This represents movement up and down in the direction of gravity by the
028     * Earth.  Zero represents a ground-level object, while an increasing value
029     * would represent moving up 
030     */
031    private double Z;
032        
033    /**
034     * Initializes a zero Vector3D
035     */
036
037    /**
038     * Initializes a new Vector3D with the given coordinates
039     * @param x the x coordinate
040     * @param y the y coordinate
041     * @param z the z coordinate
042     */
043    public Vector3D(double x, double y, double z) 
044    { 
045        this.X = x; 
046        this.Y = y; 
047        this.Z = z; 
048    }
049    
050    /** 
051     * Default no-args constructor
052     */
053    public Vector3D() 
054    { 
055        this(0.0, 0.0, 0.0);
056    }
057    
058    /**
059     * Constructs a 3D vector (with Z = 0) 
060     * @param magnitude the magnitude of the vector
061     * @param direction the direction of the vector
062     */
063    public Vector3D(double magnitude, double direction)
064    {
065        this(RobotCode2015.util.MathUtil.cos(direction)*magnitude,
066             RobotCode2015.util.MathUtil.sin(direction)*magnitude,
067             0);
068    }
069    
070    /** 
071     * Adds this Vector3D with another specified Vector3D
072     * @param v the vector to add
073     * @return a new Vector3D that is the sum of the two vectors
074     */
075    public Vector3D add(Vector3D v)
076    {
077        return new Vector3D(this.X + v.X, this.Y + v.Y, this.Z + v.Z);
078    }
079    
080    /**
081     * Subtracts this Vector3D with another specified Vector3D
082     * @param v the vector to subtract from this vector 
083     * @return a new Vector3D that is the difference of the two vectors
084     */
085    public Vector3D subtract(Vector3D v)
086    {
087        return new Vector3D(this.X - v.X, this.Y - v.Y, this.Z - v.Z);
088    }
089    
090    /**
091     * Scales the given vector by performing scalar multiplication
092     * @param d the value with which to scale this vector
093     * @return a new Vector that is the original vector scaled by value d
094     */
095    public Vector3D scale(double d)
096    {
097        return new Vector3D(this.X*d, this.Y*d, this.Z*d);
098    }
099    
100    /**
101     * Gets the magnitude of this vector
102     * @return the magnitude of this vector
103     */
104    public double magnitude()
105    {
106        return Math.sqrt(X * X + Y * Y + Z * Z);
107    }
108    
109    /**
110     * Gets the norm of this vector through (V / |V|)
111     * @return 
112     */
113    public Vector3D normalize() 
114    {
115        double mag = magnitude();
116        return new Vector3D(
117            X / mag,
118            Y / mag,
119            Z / mag
120        );
121    }
122    
123    /**
124     * Calculates the dot product of this vector with the given vector
125     * @param v the other vector
126     * @return the dot product of the two vectors
127     */
128    public double dot(Vector3D v)
129    {
130        return X * v.X + Y * v.Y + Z * v.Z;
131    }
132    
133    /**
134     * Returns the cross product of this vector with the given vector
135     * @param v the other vector
136     * @return the cross product of the two vectors
137     */
138    public Vector3D cross(Vector3D v)
139    {
140        /* a cross b formula:
141         * X = a2b3 - a3b2
142         * Y = a3b1 - a1b3
143         * Z = a1b2 - a2b1
144         */
145        
146        double x = this.Y*v.Z - this.Z*v.Y;
147        double y = this.Z*v.X - this.X*v.Z;
148        double z = this.X*v.Y - this.Y*v.X;
149        
150        return new Vector3D(x,y,z);
151    }
152    
153    /**
154     * Returns the scalar triple of three vectors
155     * @param v1 the first vector
156     * @param v2 the second vector
157     * @param v3 the third vector
158     * @return the scalar triple
159     */
160    public double scalTrip(Vector3D v1, Vector3D v2, Vector3D v3)
161    {
162        //v1 dot (v2 cross v3)
163        return v1.dot(v2.cross(v3));
164    }
165    
166    /**
167     * Returns the vector triple of three vectors
168     * @param v1 the first vector
169     * @param v2 the second vector
170     * @param v3 the third vector
171     * @return the vector triple
172     */
173    public Vector3D vectorTrip(Vector3D v1, Vector3D v2, Vector3D v3)
174    {
175        return v1.cross(v2.cross(v3));
176    }
177    
178    /**
179     * 
180     * @return 
181     */
182    public double direction() {
183        double theta = MathUtil.aTan(Y/X);
184        if (X < 0) {
185            theta = Math.PI - theta;
186        }
187        return theta;
188    }
189        
190    /**
191     * Returns a string representation of the Vector3D
192     * @return 
193     */
194    public String toString() 
195    {
196        return "<" + 
197               IOUtil.padRight(Double.toString(X), 6).substring(0, 5) + ", " + 
198               IOUtil.padRight(Double.toString(Y), 6).substring(0, 5) + ", " + 
199               IOUtil.padRight(Double.toString(Z), 6).substring(0, 5) + ">";
200    }
201}

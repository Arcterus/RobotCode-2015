001package RobotCode2015.util;
002
003/**
004 * IO util library with some utility methods for writing to console and strings in general
005 * @author Ben Huchley
006 * @author Manan Shah
007 * @author Andrew Tierno
008 */
009public class IOUtil {
010    
011    /** 
012      * Call when something minor goes wrong that shouldn't crash the robot.
013      * @param s a descriptive string about the problem
014      */
015     public static void warn(String s) {
016         System.err.print("Warn: ");
017         System.err.println(s);
018     }
019 
020    /**
021     * Pads the given string on the right with spaces
022     * @param s The string which we are padding
023     * @param length the desired final length of our string.
024     * @return 
025     */
026    public static String padRight(String s, int length) {
027        StringBuffer sb = new StringBuffer();
028        sb.append(s);
029        for(int i = s.length(); i < length; i++)
030            sb.append(" ");
031        return sb.toString();
032    }
033    
034    /**
035     * Pads the given string on the left with spaces
036     * @param s The string which we are padding
037     * @param length the desired final length of our string.
038     * @return 
039     */
040    public static String padLeft(String s, int length) {
041            StringBuffer sb = new StringBuffer();
042            for(int i = s.length(); i < length; i++)
043                sb.append(" ");
044            sb.append(s);
045            return sb.toString();
046    }
047    
048    /** 
049     * Prints a line.
050     * This is really simple, it exists in case we need to change this later
051     * so that we can change it everywhere easily
052     * @param obj the line to print
053     */
054    public static void println(Object obj) {
055        System.out.println(obj);
056    }
057    
058    public static void println(boolean b) {
059        println(String.valueOf(b));
060    }
061    
062    public static void println(char c) {
063        println(String.valueOf(c));
064    }
065    
066    public static void println(char[] s) {
067        println(String.valueOf(s));
068    }
069    
070    public static void println(double d) {
071        println(String.valueOf(d));
072    }
073    
074    public static void println(float f) {
075        println(String.valueOf(f));
076    }
077    
078    public static void println(int i) {
079        println(String.valueOf(i));
080    }
081    
082    public static void println(long l) {
083        println(String.valueOf(l));
084    }
085    
086}

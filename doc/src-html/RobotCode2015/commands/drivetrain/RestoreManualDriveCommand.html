001package RobotCode2015.commands.drivetrain;
002
003import RobotCode2015.OI;
004import RobotCode2015.subsystems.Drivetrain;
005import RobotCode2015.Constants;
006import RobotCode2015.commands.CommandBase;
007
008/**
009 *
010 * @author Brian Chan
011 */
012public class RestoreManualDriveCommand extends CommandBase {
013    
014    private boolean hasReset;
015    
016    public RestoreManualDriveCommand() {
017    }
018    
019    public void start() {
020        super.start();
021        hasReset = false;
022    }
023
024    // Called just before this Command runs the first time
025    protected void initialize() {
026    }
027
028    // Called repeatedly when this Command is scheduled to run
029    protected void execute() {
030        if(checkAxesMinLessThan(Constants.Drivetrain.manualOverrideResetThreshold))
031            hasReset = true;
032    }
033
034    // Make this return true when this Command no longer needs to run execute()
035    protected boolean isFinished() {
036        return !drivetrain.isBusy() || (hasReset && checkAxesMaxGreaterThan(Constants.Drivetrain.manualOverrideThreshold));
037    }
038
039    // Called once after isFinished returns true
040    protected void end() {
041        drivetrain.setBusy(false);
042        OI.manualDrive.start();
043    }
044
045    // Called when another command which requires one or more of the same
046    // subsystems is scheduled to run
047    protected void interrupted() {
048    }
049    
050    private boolean checkAxesMaxGreaterThan(double limit) {
051        return (Math.abs(RobotCode2015.OI.gamepad1.getLeftX()) > limit || Math.abs(RobotCode2015.OI.gamepad1.getLeftY()) > limit || 
052                Math.abs(RobotCode2015.OI.gamepad1.getRightX()) > limit || Math.abs(RobotCode2015.OI.gamepad1.getRightY()) > limit);
053    }
054    
055    private boolean checkAxesMinLessThan(double limit) {
056        return (Math.abs(RobotCode2015.OI.gamepad1.getLeftX()) < limit || Math.abs(RobotCode2015.OI.gamepad1.getLeftY()) < limit || 
057                Math.abs(RobotCode2015.OI.gamepad1.getRightX()) < limit || Math.abs(RobotCode2015.OI.gamepad1.getRightY()) < limit);
058    }
059}

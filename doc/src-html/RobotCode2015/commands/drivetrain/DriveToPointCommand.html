001package RobotCode2015.commands.drivetrain;
002import RobotCode2015.util.MathUtil;
003import com.sun.squawk.util.MathUtils;
004import edu.wpi.first.wpilibj.Timer;
005
006/**
007 * Drives to a point (x,y) at a given speed.
008 * @author neymikajain
009 */
010public class DriveToPointCommand extends AutomaticDriveCommand {
011    
012    public double x, y, angle, speed;
013    public double distance;
014    public double start;
015    public boolean isCommandFinished;
016    
017    public DriveToPointCommand(double x, double y, double speed) {
018        super ("Drive to Point Command");
019        requires(drivetrain);
020        this.x = x;
021        this.y = y;
022        angle = MathUtils.atan((x/y));
023        this.distance = 0;
024        start = drivetrain.getAverageDistance();
025
026        this.speed = speed;
027    }
028    
029    
030    protected void initialize() {
031        
032    }
033
034    // Called repeatedly when this Command is scheduled to run
035    protected void execute() 
036    {       
037        isCommandFinished = drivetrain.driveToPoint(x, y, speed, Timer.getFPGATimestamp() - startTime);
038    }
039
040    // Make this return true when this Command no longer needs to run execute()
041    protected boolean isFinished() 
042    {
043        if(((drivetrain.getAverageDistance() - start) >= distance) &&  drivetrain.getYawAngle() == angle) 
044            return true; //This scenario is unrealistic, it returns true only if the robot is EXACTLY at the desired point. Might want to use a threshold?
045        return isCommandFinished;
046    }
047}

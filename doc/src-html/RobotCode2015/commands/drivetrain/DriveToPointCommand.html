008/**
009 *
010 * @author neymikajain
011 */
012public class DriveToPointCommand extends AutomaticDriveCommand {
013    
014    public double x, y, angle, speed;
015    public double distance;
016    public double start;
017    public long startTime;
018    public boolean isCommandFinished;
019    
020    public DriveToPointCommand(double x, double y, double speed) 
021    {
022        requires(drivetrain);
023        this.x = x;
024        this.y = y;
025        angle = MathUtil.aTan((x/y));
026        this.distance = 0;
027        start = drivetrain.getAverageDistance();
028
029        this.speed = speed;
030    }
031
032    // Called just before this Command runs the first time
033    protected void initialize() {
034        startTime = System.currentTimeMillis();
035        isCommandFinished = false;
036    }
037
038    // Called repeatedly when this Command is scheduled to run
039    protected void execute() 
040    {       
041        isCommandFinished = drivetrain.driveToPoint(x, y, speed, System.currentTimeMillis() - startTime);
042    }
043
044    // Make this return true when this Command no longer needs to run execute()
045    protected boolean isFinished() 
046    {
047        if(((drivetrain.getAverageDistance() - start) >= distance) &&  drivetrain.getYawAngle() == angle)
048            return true; 
049        return isCommandFinished;
050    }
051}

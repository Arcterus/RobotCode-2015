005package RobotCode2015.commands.drivetrain;
006
007import RobotCode2015.OI;
008import RobotCode2015.commands.CommandBase;
009
010/**
011 * Framework that should be inherited by any Command that autonomously utilizes the drivetrain.
012 * @author Manan
013 */
014public abstract class AutomaticDriveCommand extends CommandBase {
015    
016    public AutomaticDriveCommand(String name) {
017        super (name);
018    }
019    
020    // Called once after isFinished returns true
021    protected void end() {
022        super.end();
023        OI.manualDriveRestore.start();
024        drivetrain.stop();
025    }
026
027    // Called when another command which requires one or more of the same
028    // subsystems is scheduled to run
029    protected void interrupted() {
030        end();
031    }
032}

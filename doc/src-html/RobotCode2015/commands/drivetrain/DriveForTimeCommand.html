005package RobotCode2015.commands.drivetrain;
006import RobotCode2015.commands.CommandBase;
007import edu.wpi.first.wpilibj.Timer;
008
009/**
010 *
011 * @author Manan
012 */
013public class DriveForTimeCommand extends AutomaticDriveCommand {
014    private final double left, right;
015    private boolean isStopped;
016    
017    public DriveForTimeCommand(double left, double right, double timeout) {
018        requires(drivetrain);
019        this.right = right;
020        this.left = left;
021        isStopped = true;
022        setTimerLength(timeout);
023    }
024
025    // Called just before this Command runs the first time
026    protected void initialize() {
027    }
028
029    // Called repeatedly when this Command is scheduled to run
030    protected void execute() {
031        if(isStopped) {
032            System.out.println("just started");
033            isStopped = false;
034            startTime = Timer.getFPGATimestamp(); //hack to restart timer
035        }
036        
037        drivetrain.drive(left, right);
038    }
039
040    // Make this return true when this Command no longer needs to run execute()
041    protected boolean isFinished() {
042        return isTimerFinished();
043    }
044
045    // Called once after isFinished returns true
046    protected void end() {
047        System.out.println("finished driving");
048        isStopped = true;
049    }
050
051    // Called when another command which requires one or more of the same
052    // subsystems is scheduled to run
053    protected void interrupted() {
054        end();
055    }
056}
